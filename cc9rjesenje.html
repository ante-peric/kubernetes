9.7. Zadatak

vim ljubica.yml

apiVersion: apps/v1
kind: ReplicaSet
metadata:
    name: ljubica
    labels:
      type: jsp
      author: Ante-Peric

spec:
  template:
   metadata:
    name: ljubica-pod
    labels:
     type: jsp
     author: Ante-Peric
   spec:
    containers:
     - name: nginx-container
      image: nginx
  replicas: 3
  selector:
    matchLabels:
      type: jsp
      author: Ante-Peric

kubectl create -f ljubica.yml

kubectl expose replicaset ljubica --type=NodePort --port=80 --target-port=80
